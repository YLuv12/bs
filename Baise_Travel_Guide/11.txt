-- ====================== 1. 管理员表（新增，适配后台管理） ======================
DROP TABLE IF EXISTS `admin`;

  `id` bigint NOT NULL AUTO_INCREMENT COMMENT '管理员ID',

  `password` varchar(100) NOT NULL COMMENT '密码（bcrypt加密）',

  `phone` varchar(20) DEFAULT NULL COMMENT '联系方式',

  `create_time` datetime DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `update_time` datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  `delete_time` datetime DEFAULT NULL COMMENT '逻辑删除时间（NULL=未删除，非NULL=已删除）',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_admin_username` (`username`),
  KEY `idx_delete_time` (`delete_time`)

CREATE TABLE `sys_dict` (
  `id` bigint NOT NULL AUTO_INCREMENT COMMENT '字典ID',
  `dict_type` varchar(50) NOT NULL COMMENT '字典类型（如：spot_type、food_type、feedback_type）',
  `dict_code` varchar(50) NOT NULL COMMENT '字典编码（枚举值）',
  `dict_label_cn` varchar(100) NOT NULL COMMENT '字典标签（中文）',
  `dict_label_en` varchar(200) NOT NULL COMMENT '字典标签（英文）',
  `sort` int NOT NULL DEFAULT 0 COMMENT '排序值',
  `status` tinyint NOT NULL DEFAULT 1 COMMENT '状态：1-启用 0-禁用',
  `create_time` datetime DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `update_time` datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  `delete_time` datetime DEFAULT NULL COMMENT '逻辑删除时间（NULL=未删除，非NULL=已删除）',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_dict_type_code` (`dict_type`,`dict_code`),
  KEY `idx_dict_type` (`dict_type`),
  KEY `idx_status` (`status`),
  KEY `idx_delete_time` (`delete_time`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='系统数据字典表（统一管理枚举类）';
DROP TABLE IF EXISTS `sys_config`;
CREATE TABLE `sys_config` (
  `id` bigint NOT NULL AUTO_INCREMENT COMMENT '配置ID',
  `config_key` varchar(100) NOT NULL COMMENT '配置键（如：upload_max_size、comment_audit_switch）',
  `config_value` text NOT NULL COMMENT '配置值（如：1024000、true）',
  `config_name` varchar(100) NOT NULL COMMENT '配置名称（如：文件上传最大大小、评论审核开关）',
  `config_type` varchar(50) DEFAULT 'string' COMMENT '配置类型（string/int/boolean）',
  `module` varchar(50) DEFAULT NULL COMMENT '所属模块（如：系统设置、内容管理）',
  `remark` varchar(200) DEFAULT NULL COMMENT '备注',
  `create_time` datetime DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `update_time` datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  `delete_time` datetime DEFAULT NULL COMMENT '逻辑删除时间（NULL=未删除，非NULL=已删除）',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_config_key` (`config_key`),
  KEY `idx_module` (`module`),
  KEY `idx_delete_time` (`delete_time`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='系统参数配置表';
